from datetime import datetime from js import document def get_current_time(event): # Get the current date and time now = datetime.now() current_date = now.strftime("%Y-%m-%d") current_time = now.strftime("%H:%M") # Update the current time paragraph document.getElementById('current-time').innerText = f"Current Date and Time: {now.strftime('%Y-%m-%d %H:%M:%S')}" # Set the current date and time in the input fields document.getElementById('date').value = current_date document.getElementById('time').value = current_time # Attach the calculation function to the button click document.getElementById("current-time-button").onclick = get_current_time
